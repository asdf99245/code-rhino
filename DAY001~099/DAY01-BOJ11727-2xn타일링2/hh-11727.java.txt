import java.util.Scanner;

public class Main {
	public static void main(String[] args) {
		Scanner scan = new Scanner(System.in);
		int n = scan.nextInt();
		int[] dp = new int[1001];
		int mod = 10007;
		
		dp[1] = 1;
		for(int i=2; i<=n; i++) {
			if(i%2==0) dp[i] = (dp[i-1] * 2 + 1) % mod;
			else dp[i] = (dp[i-1] * 2 - 1) % mod;
		}
		
		System.out.println(dp[n]);
	}
}
